============================================================
 SQLite 数据库 (epmc_articles.db) 常用管理命令
============================================================
-- 表名: articles
-- 关键字段: 
--   epmc_id (主键), pmid, doi, title, abstract, 
--   pub_year (年份), author_string, journal_title (期刊), 
--   source (来源, 如 'MED' 或 'PPR'), query_search_term (搜索词)
============================================================


--- 1. 查找和浏览 (SELECT) ---

-- 查看表中的 10 条数据 (所有字段)
-- * 号代表所有字段，LIMIT 10 限制只返回10条，避免数据太多卡住
SELECT * FROM articles LIMIT 10;

-- 只查看特定的几个字段 (例如: 标题, 年份, 期刊)
SELECT title, pub_year, journal_title FROM articles LIMIT 20;


--- 2. 筛选 (WHERE) ---

[cite_start]-- 筛选特定年份的文章 (例如: 2025年) [cite: 1, 10, 23, 46]
SELECT title, pub_year FROM articles WHERE pub_year = 2025;

-- 筛选一个年份范围内的文章 (例如: 2024年至今)
SELECT title, pub_year FROM articles WHERE pub_year >= 2024;

[cite_start]-- 筛选特定期刊的文章 (例如: Nature microbiology) [cite: 53]
-- 注意: 'Nature microbiology' 必须和你数据库中存储的完全一样，区分大小写
SELECT title, pub_year, author_string FROM articles WHERE journal_title = 'Nature microbiology';

[cite_start]-- 筛选所有预印本 (Preprints) [cite: 10, 23]
-- 假设预印本的 source 字段是 'PPR'
SELECT title, source FROM articles WHERE source = 'PPR';

[cite_start]-- 筛选所有正式发表的 (例如: MEDLINE) [cite: 1, 46, 63]
SELECT title, source FROM articles WHERE source = 'MED';

[cite_start]-- 在标题 (title) 中搜索关键词 (例如: 包含 "CRISPR") [cite: 210, 222]
-- % 是通配符，代表任意字符。%CRISPR% 表示包含 "CRISPR" 即可
SELECT title, pub_year FROM articles WHERE title LIKE '%CRISPR%';

[cite_start]-- 在摘要 (abstract) 中搜索关键词 (例如: 包含 "microbiome") [cite: 3, 25]
SELECT abstract, title FROM articles WHERE abstract LIKE '%microbiome%';

[cite_start]-- 组合筛选：在摘要中同时包含 "microbiome" 和 "LLM" 的文章 [cite: 28, 37]
SELECT title, abstract FROM articles 
WHERE abstract LIKE '%microbiome%' AND abstract LIKE '%LLM%';

[cite_start]-- 组合筛选：标题包含 "Alzheimer" 或者 "Parkinson" 的文章 [cite: 128, 137]
SELECT title FROM articles 
WHERE title LIKE '%Alzheimer%' OR title LIKE '%Parkinson%';

[cite_start]-- 按特定ID精确查找 (例如: pmid) [cite: 1]
SELECT * FROM articles WHERE pmid = '41127671';

-- 查找某个特定搜索词 (query_search_term) 导入的所有文章
[cite_start]-- (你需要把 ... 换成你当时搜索用的实际关键词) [cite: 9, 23]
SELECT epmc_id, title FROM articles WHERE query_search_term = '(microbiome OR microbiota OR "microbial community" OR metagenome OR metagenomics) AND ("large language model" OR "foundation model")';


--- 3. 排序 (ORDER BY) ---

-- 查找所有文章，按出版年份倒序 (最新的在最前)
SELECT title, pub_year FROM articles 
ORDER BY pub_year DESC
LIMIT 50; -- 只看最新的50篇

-- 查找所有文章，按期刊名称字母顺序排序
SELECT title, journal_title FROM articles 
ORDER BY journal_title ASC
LIMIT 50;


--- 4. 计数 (COUNT) ---

-- 计算数据库中总共有多少篇文章
SELECT COUNT(*) FROM articles;

-- 计算每个年份分别有多少篇文章
SELECT pub_year, COUNT(*) 
FROM articles 
GROUP BY pub_year 
ORDER BY pub_year DESC;

-- 计算每个期刊分别有多少篇文章，并按文章数倒序
SELECT journal_title, COUNT(*) 
FROM articles 
GROUP BY journal_title 
ORDER BY COUNT(*) DESC;

[cite_start]-- 计算来源是 'PPR' (预印本) 的有多少篇 [cite: 10, 23]
SELECT COUNT(*) FROM articles WHERE source = 'PPR';
